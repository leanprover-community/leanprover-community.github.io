Mario Carneiro (Sep 02 2018 at 19:26):

import algebra.group_power

theorem th01 (a m n : ℕ) : a ^ (m + n) = (a ^ m) * (a ^ n) :=
by simpa using pow_add a m n

Mario Carneiro (Sep 02 2018 at 19:27):

theorem th01 (a m n : ℕ) : a ^ (m + n) = (a ^ m) * (a ^ n) :=
by rw nat.pow_add

Mario Carneiro (Sep 02 2018 at 19:27):

theorem th01 (a m n : ℕ) : a ^ (m + n) = (a ^ m) * (a ^ n) :=
nat.pow_add _ _ _

Mario Carneiro (Sep 02 2018 at 19:28):
