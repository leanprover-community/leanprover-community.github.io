Johan Commelin (Feb 13 2019 at 13:36):

Voila:

namespace tactic
namespace interactive
open interactive interactive.types

meta def use_this (l : parse pexpr_list_or_texpr) : tactic unit :=
(tactic.use l >> (triv <|> try `[rw exists_prop]))

end interactive
end tactic
