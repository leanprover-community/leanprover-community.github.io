Mario Carneiro (Apr 18 2019 at 17:49):

def P : ∀(n : with_top ℕ), Prop
| (some n) := true
| none := ∀ n : ℕ,
    have (n : with_top ℕ) < ⊤ := with_top.coe_lt_top n,
    P n
using_well_founded { rel_tac := λ_ _, `[exact ⟨(<), with_top.well_founded_lt nat.lt_wf⟩],
dec_tac := tactic.assumption }
