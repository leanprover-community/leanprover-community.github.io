Inspired by a recent numberphile video, here's Graham's number in lean:

-- knuth e m n = m ^(e) n
def knuth : ℕ → ℕ → ℕ → ℕ
| 0     m n := m * n
| (e+1) m n := (knuth e m)^[n-1] m

def g : ℕ → ℕ
| 0 := 4
| (n+1) := knuth (g n) 3 3

def graham_number := g 64

#eval graham_number
