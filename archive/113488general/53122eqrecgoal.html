-- value groups.

(minimal_value_group v₁).Γ  → (minimal_value_group v₂).Γ :=
λ g, begin

    exact finsupp.prod g (λ r n,(minimal_value_group.mk v₂ r) ^ n),
  cases h12 with h12 hoops,
