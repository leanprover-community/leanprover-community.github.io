Scott Morrison (Jun 27 2019 at 01:27):

adding the lemma

@[simp] lemma nat.cast_with_bot_cast {α} [add_monoid α] [has_one α] : ∀ n : ℕ,
  (n : with_bot α) = (n : α)
| 0 := rfl
| (n+1) := by simp [nat.cast_with_bot_cast n]

example (n : ℕ) : degree (X^n : polynomial α) = n :=
by simp

works
