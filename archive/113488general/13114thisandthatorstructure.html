def topological_space.open_immersion {X Y : Type} [tX : topological_space X] [tY : topological_space Y] (φ : X → Y) :=
  continuous φ ∧
  function.injective φ ∧
  ∀ U : set X, tX.is_open U → tY.is_open (set.image φ U)
