import tactic.interactive

lemma misc_tauto_a (p q : Prop) : (p ↔ p ∧ q) ↔ (p → q) :=
begin
  cases (classical.em p);
  cases (classical.em q),
  repeat {tauto},
  /- one goal lives on!
  p q : Prop,
  h : p,
  h_1 : ¬q
