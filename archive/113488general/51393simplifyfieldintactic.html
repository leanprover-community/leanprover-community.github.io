Kenny Lau (Sep 10 2018 at 06:05):

my workaround:

structure my_struct :=
(f : ℕ)

example : false := begin
  let s : my_struct := ⟨42⟩,
  let v := s.f,
  have hv : v = 42 := rfl,
  admit
end
