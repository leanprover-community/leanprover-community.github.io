Mario Carneiro (Jun 28 2019 at 09:47):

def mth (A : Type) : ∀ (n : ℕ), vec A n → xfin n → A
| _ (vec.cons n a l) (xfin.zero _) := a
| _ (vec.cons _ a l) (xfin.succ n i) := mth n l i
