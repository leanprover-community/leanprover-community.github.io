/-- If `x ∣ y` return an `expr` containing a proof! -/
meta def baz (x y : ℕ) : tactic (option expr) :=
if x ∣ y then do
  (_, p) ← norm_num.derive `((%%(reflect x) : ℕ) ∣ %%(reflect y)),
  p' ← tactic.mk_eq_mpr p `(trivial),
  return (some p')
else
pure none
