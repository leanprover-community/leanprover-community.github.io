Kenny Lau (Mar 16 2019 at 10:42):

import data.int.basic

instance asdf {α : Type*} [add_group α] [has_one α] : is_add_monoid_hom (int.cast : ℤ → α) :=
⟨int.cast_zero, int.cast_add⟩
