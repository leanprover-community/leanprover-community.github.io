def evv (x :ℕ → ℕ) (y: ℕ):=
    match x y with
    | 0 := tt
    | 1 := ff
    | _ := tt
    end

theorem test: ∀ x (f: ℕ → ℕ), f x=1 → evv f x :=
begin
    intros, unfold evv, ...

At the point of the "...", I have the following state:

x : ℕ,
f : ℕ → ℕ,
a : f x = 1
⊢ ↥(evv._match_1 (f x))

How do I complete the theorem?
