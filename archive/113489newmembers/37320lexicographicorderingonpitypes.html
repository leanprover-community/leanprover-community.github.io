variables {α : Type*} {β :  α → Type*} (r : α → α → Prop)  (s : Π {a : α}, β a → β a → Prop)

def pi.lex (f g : Π a, β a) : Prop :=
∃ a, (∀ b, r b a → f b = g b) ∧ s (f a) (g a)
