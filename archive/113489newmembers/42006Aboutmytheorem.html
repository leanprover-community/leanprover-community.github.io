def fin_n_to_list {n : ℕ} (x: fin n → ℕ) : list ℕ :=
    (list.range n).map (λ i, if h : i < n then x ⟨i, h⟩ else 0)

def n_lcm {n : ℕ} (x : fin n → ℕ) : ℕ :=
    list.foldl nat.lcm 1 (fin_n_to_list x)

lemma n_lcm_geq_1 {n : ℕ} (x : fin n → ℕ) (hx : ∀ i, x i ≥ 1):
    n_lcm x ≥ 1 := sorry
