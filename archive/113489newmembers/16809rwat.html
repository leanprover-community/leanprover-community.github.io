Patrick Massot (Aug 11 2018 at 22:15):

import tactic.ring

example {α : Type*} [comm_ring α] {a b c d : α} :  a * b + c * d = b * a + d * c :=
by ring

example {α : Type*} [comm_ring α] {a b c d : α} :  a * b + c * d = b * a + d * c :=
begin
  rw mul_comm,
  rw add_comm,
  rw mul_comm,
  rw add_comm
end

example {α : Type*} [comm_ring α] {a b c d : α} :  a * b + c * d = b * a + d * c :=
begin
  conv {
      to_lhs,
      congr,
      rw mul_comm,
      skip,
      rw mul_comm,
  }
end

example {α : Type*} [comm_ring α] {a b c d : α} :  a * b + c * d = b * a + d * c :=
begin
  conv {
      for (a*b) [1] { rw mul_comm },
      for (c*d) [1] { rw mul_comm },
  }
end
