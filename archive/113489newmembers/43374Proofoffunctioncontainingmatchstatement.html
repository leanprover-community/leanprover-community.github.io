Kenny Lau (Jun 27 2019 at 21:45):

def f (n : ℕ) : ℕ :=
match n with
| 0 := 1
| (k+1) := 0
end

example (hf : f 0 = 2) : true :=
begin
  change 1 = 2 at hf,
  guard_hyp hf := 1 = 2,
  trivial
end
