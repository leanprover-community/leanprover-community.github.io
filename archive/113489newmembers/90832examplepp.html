Johan Commelin (Oct 22 2019 at 06:03):

Here's my version:

example (p : Prop) : ¬(p ↔ ¬p) :=
begin
  assume H,
  have hp : p := H.mpr (λ hp, _),
  all_goals
  { have hp_copy := hp,
    rw H at hp_copy,
    contradiction },
end
