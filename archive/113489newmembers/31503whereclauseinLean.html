Mario Carneiro (Jun 09 2019 at 23:32):

here's some hackery:

def reverse {α} (l : list α) : list α :=
let aux : list α → list α :=
  match l with
  | [], r := r
  | (a::l), r := by exact _match l (a::r)
  end
in aux []

#eval reverse [1,2,3]
