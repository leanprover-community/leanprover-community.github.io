import data.real.basic tactic.norm_num

example (h : (5:ℕ) = 4) : false := by cases h
example (h : (5:ℤ) = 4) : false := by cases h
example (h : (5:ℚ) = 4) : false := by norm_num at h
example (h : (5:ℝ) = 4) : false := by norm_num at h
