Hopefully there is something less hideous than

lemma int.succ_ne_self {x : ℤ} : x + 1 ≠ x :=
begin
  intro h,
  replace h := congr_arg (λ n : ℤ, n - x) h,
  dsimp at h,
  rw [add_assoc, add_comm, neg_add_cancel_right, add_right_neg] at h,
  exact one_ne_zero h
end
