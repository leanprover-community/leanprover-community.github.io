Well here's a tactic mode proof:

example (p q r : Prop) : ((p ∨ q) → r) ↔ (p → r) ∧ (q → r) :=
begin
  split,
  { intro H,
    split,
    { intro Hp,
      apply H,
      left,
      exact Hp
    },
    { intro Hq,
      apply H,
      right,
      exact Hq
    }
  },
  intro H,
  cases H with Hpr Hqr,
  intro Hpq,
  cases Hpq with Hp Hq,
  { apply Hpr,
    exact Hp
  },
  { apply Hqr,
    exact Hq
  }
end
